02 Normalisation — Why 3NF Phase 1
1NF
First Normal Form
  • Each column holds a single atomic value
  • No comma-separated option lists in Questions
  • Options A/B/C/D stored in separate columns
  • No arrays — UserAnswers is a separate table
2NF
Second Normal Form
  • All non-key columns fully depend on PK
  • Question text not stored in UserAnswers — FK reference only
  • Paper metadata not duplicated in Sections
  • Exam info not repeated in every MockTest row
3NF
Third Normal Form
  • No transitive dependencies
  • MarksPerQuestion stored in Section, not repeated in UserAnswers
  • ExamCode does NOT determine TotalMarks (MockTest does)
  • Percentage in UserAttempts is computed but stored for query perf
DEN
Controlled Denormalisation
  • TotalScore, CorrectCount, Percentage stored in UserAttempts
  • Calculated at submission time, cached for fast reads
  • Avoids expensive SUM() on every result screen load
  • Acceptable: source data (UserAnswers) always exists for recalc
03 Table Relationships Phase 1
Exams → MockTests one exam has many mock tests (Prelim/Mains papers) 1 : N
MockTests → Sections each paper divided into subject sections 1 : N
Sections → Questions each section owns its questions 1 : N
MockTests → UserAttempts many users attempt each mock test 1 : N
Users → UserAttempts each user has multiple attempts across tests 1 : N
UserAttempts → UserAnswers one attempt has one answer row per question 1 : N
Questions → UserAnswers each question answered by many users 1 : N

Configurations/MockTestConfiguration.cs C# 
public class MockTestConfiguration : IEntityTypeConfiguration<MockTest>
{
    public void Configure(EntityTypeBuilder<MockTest> builder)
    {
        builder.ToTable("MockTests");
        builder.HasKey(x => x.MockTestId);

        builder.Property(x => x.Title)
               .IsRequired().HasMaxLength(200);

        builder.Property(x => x.PaperType)
               .IsRequired()
               .HasConversion<string>()    // stores "Prelim"/"Mains" as string
               .HasMaxLength(20);

        builder.Property(x => x.PassingMarks)
               .HasColumnType("decimal(6,2)");

        builder.Property(x => x.CreatedAt)
               .HasDefaultValueSql("GETUTCDATE()");

        builder.HasOne(x => x.Exam)
               .WithMany(e => e.MockTests)
               .HasForeignKey(x => x.ExamId)
               .OnDelete(DeleteBehavior.Restrict);

        builder.HasIndex(x => x.ExamId);
    }
}
Configurations/UserAttemptConfiguration.cs C# 
public class UserAttemptConfiguration : IEntityTypeConfiguration<UserAttempt>
{
    public void Configure(EntityTypeBuilder<UserAttempt> builder)
    {
        builder.ToTable("UserAttempts");
        builder.HasKey(x => x.AttemptId);

        builder.Property(x => x.Status)
               .HasConversion<string>()
               .HasMaxLength(20);

        builder.Property(x => x.TotalScore).HasColumnType("decimal(8,2)");
        builder.Property(x => x.Percentage).HasColumnType("decimal(5,2)");

        // One attempt per user per test (relaxed with a unique index)
        builder.HasIndex(x => new { x.UserId, x.MockTestId })
               .IsUnique(false);  // set true if you want 1-attempt limit

        builder.HasOne(x => x.MockTest)
               .WithMany(m => m.UserAttempts)
               .HasForeignKey(x => x.MockTestId)
               .OnDelete(DeleteBehavior.Restrict);

        builder.HasIndex(x => x.UserId);
        builder.HasIndex(x => x.MockTestId);
    }
}
Configurations/UserAnswerConfiguration.cs C# 
public class UserAnswerConfiguration : IEntityTypeConfiguration<UserAnswer>
{
    public void Configure(EntityTypeBuilder<UserAnswer> builder)
    {
        builder.ToTable("UserAnswers");
        builder.HasKey(x => x.AnswerId);

        builder.Property(x => x.SelectedOption).HasMaxLength(1);
        builder.Property(x => x.MarksAwarded).HasColumnType("decimal(6,2)");

        // One answer row per question per attempt
        builder.HasIndex(x => new { x.AttemptId, x.QuestionId }).IsUnique();

        builder.HasOne(x => x.UserAttempt)
               .WithMany(a => a.UserAnswers)
               .HasForeignKey(x => x.AttemptId)
               .OnDelete(DeleteBehavior.Cascade); // delete answers with attempt

        builder.HasOne(x => x.Question)
               .WithMany(q => q.UserAnswers)
               .HasForeignKey(x => x.QuestionId)
               .OnDelete(DeleteBehavior.Restrict);
    }
}

💡 Scoring logic in AttemptService.SubmitAttemptAsync:
For each answer: load Question (with SectionId) → load Section (MarksPerQuestion, NegativeMarks)
If SelectedOption == CorrectOption → MarksAwarded = MarksPerQuestion
If SelectedOption != null && != CorrectOption → MarksAwarded = -NegativeMarks
If SelectedOption == null → MarksAwarded = 0 (skipped)
TotalScore = SUM(MarksAwarded), Percentage = (TotalScore / TotalMarks) × 100
